                let servicesArray = [];
                if (Array.isArray(data)) servicesArray = data;
                else if (data.data && Array.isArray(data.data)) servicesArray = data.data;
                else if (data.services && Array.isArray(data.services)) servicesArray = data.services;
                else if (data.result && Array.isArray(data.result)) servicesArray = data.result;
                else if (data.items && Array.isArray(data.items)) servicesArray = data.items;
                else {
                    // วนลูปหา Array แรกที่เจอใน Object
                    for (let key in data) {
                        if (Array.isArray(data[key])) {
                            servicesArray = data[key];
                            break;
                        }
                    }
                }
